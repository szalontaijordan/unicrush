3   import java.util.ArrayList;
4   import java.util.List;
5   
6   /**
7    * Class representing the game like Candy Crush with a list of levels you can
8    * play.
9    *
10   * @author Szalontai Jord√°n
11   */
12  public class UniGame implements Game {
13  
14      /**
15       * The maximum amount of iterations that can occur during processing a
16       * {@code Level} of the {@code UniGame}.
17       */
18      public static final int MAX_ITERATION = 50;
19  
20      private long playerScore;
21      private int currentLevel;
22      private List<Level> levels;
23  
24      /**
25       * Constructs a {@code Game} and setting its current level to the given
26       * parameter.
27       *
28       * @param currentLevel an index of the {@code Level}
29       */
30      public UniGame(int currentLevel) {
31          this.playerScore = 0;
32          this.levels = new ArrayList<>();
33          this.currentLevel = currentLevel;
34      }
35  
36      public List<Level> getLevels() {
37          return levels;
38      }
39  
40      /**
41       * Initializes a {@code Level} so there are not any matching {@code Candy}
42       * sequences.
43       *
44       * @param levelIndex the index of the {@code Level} we'd like to start
45       *
46       * @see LevelManager#processLevel
47       * @see LevelManager#reset
48       */
49      public void startLevel(int levelIndex) {
50          Level level = levels.get(levelIndex);
51  
52          System.out.println("Preprocessing");
53  
54          int iterations = LevelManager.processLevel(level);
55  
56          if (iterations == MAX_ITERATION) {
57              System.out.println("Maxit, preprocessing again");
58              LevelManager.reset(level);
59              LevelManager.processLevel(level);
60          }
61  
62          System.out.println("Start:");
63          System.out.println(level);
64  
65          this.playerScore = 0;
66      }
67  
68      public void setCurrentLevel(int newCurrent) {
69          this.currentLevel = newCurrent;
70      }
71  
72      @Override
73      public Level getLevel(int index) {
74          return levels.get(index);
75      }
76  
77      @Override
78      public long getPlayerScore() {
79          return playerScore;
80      }
81  
82      @Override
83      public int getCurrent() {
84          return currentLevel;
85      }
86  
87      @Override
88      public void startCurrentLevel() {
89          startLevel(currentLevel);
90      }
91  
92      @Override
93      public void addToScore(long plus) {
94          this.playerScore += plus;
95      }
96  
97      @Override
98      public void initLevels() throws Exception {
99          // creating level, we didn't set up any DB connection yet
100         Level l = new StandardLevelBuilder()
101                 .setCompliteScore(500)
102                 .setAvailableSteps(20)
103                 .setBoardSize(8)
104                 .putWallsFromString("0,0;1,0;6,0;7,0;0,7;1,7;6,7;7,7")
105                 .fillBoard()
106                 .create();
107 
108         levels.add(l);
109     }
110 
111     @Override
112     public String toString() {
113         return "gg";
114     }
115 }
