5   /**
6    * Class for building a {@code StandardLevel}.
7    *
8    * This class is a subclass of {@code LevelBuilder}
9    *
10   * @author Szalontai Jord√°n
11   */
12  public class StandardLevelBuilder extends LevelBuilder {
13  
14      /**
15       * Constructs a {@code StandardLevelBuilder} that has a new
16       * {@code StandardLevel} field which we build.
17       */
18      public StandardLevelBuilder() {
19          super();
20          this.level = new StandardLevel();
21      }
22  
23      /**
24       * Constructs a {@code StandardLevelBuilder} that has a copy of an existing
25       * {@code StandardLevel} field which we continue building.
26       *
27       * @param l the {@code Level} which we continue building
28       */
29      public StandardLevelBuilder(Level l) {
30          super(l);
31      }
32  
33      /**
34       * Fills up the board with random {@code Candy} instances and the given
35       * walls.
36       *
37       * @return {@code this} so we can chain the methods after each other
38       * @throws Exception if something unexpected happens, e.g. the walls are not
39       * set
40       */
41      @Override
42      public LevelBuilder fillBoard() throws Exception {
43          if (level.getBoardSize() == 0) {
44              throw new Exception("Board size must be set first!");
45          }
46  
47          if (level.getWalls() == null) {
48              throw new Exception("Walls must be set first!");
49          }
50  
51          if (level.getBoard() == null) {
52              level.setBoard(new Candy[level.getBoardSize()][level.getBoardSize()]);
53          }
54  
55          for (int i = 0; i < level.getBoardSize(); i++) {
56              for (int j = 0; j < level.getBoardSize(); j++) {
57                  level.set(i, j, new Candy(Candy.getRandomColorState()));
58              }
59          }
60  
61          Arrays.stream(level.getWalls())
62                  .forEach(wall -> level.set(wall[0], wall[1], null));
63  
64          return this;
65      }
66  }
